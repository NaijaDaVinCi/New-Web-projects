    function startCountdown() {
      const bride = document.getElementById('bride').value.trim();
      const groom = document.getElementById('groom').value.trim();
      const date = document.getElementById('date').value;
      if (!bride || !groom || !date) return alert("Please fill all fields");
      const url = `${location.origin}${location.pathname}?bride=${encodeURIComponent(bride)}&groom=${encodeURIComponent(groom)}&date=${date}`;
      window.location.href = url;
    }

    function runTimer(targetDate) {
      const timerEl = document.getElementById('timer');
      const interval = setInterval(() => {
        const now = new Date().getTime();
        const diff = new Date(targetDate).getTime() - now;
        if (diff <= 0) {
          clearInterval(interval);
          timerEl.innerText = "üéâ It's your wedding day!";
          return;
        }
        const days = Math.floor(diff / (1000 * 60 * 60 * 24));
        const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
        const mins = Math.floor((diff / (1000 * 60)) % 60);
        const secs = Math.floor((diff / 1000) % 60);
        timerEl.innerText = `${days}d ${hours}h ${mins}m ${secs}s`;
      }, 1000);
    }

    // Load from URL
    const params = new URLSearchParams(window.location.search);
    const bride = params.get("bride");
    const groom = params.get("groom");
    const date = params.get("date");
    if (bride && groom && date) {
      document.getElementById("form-section").classList.add("hidden");
      document.getElementById("countdown-section").classList.remove("hidden");
      document.getElementById("names").innerText = `${bride} ‚ù§Ô∏è ${groom}`;
      runTimer(date);
      document.getElementById("shareLink").href = window.location.href;
    }
